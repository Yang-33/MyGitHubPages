    template<typename type>
    struct BIT0 {
        int N; int nn; vector<type> data;
        BIT0(int n) { N = n + 1; data = vector<type>(n + 1, 0); nn = 1;    while (nn * 2 <= N)nn *= 2; }
        void add(int i, type w) { i++;   for (int x = i; x < N; x += x & -x) { data[x] += w; } }
        type sum(int i) { type ret = 0;   for (int x = i; x > 0; x -= x & -x) { ret += data[x]; }  return ret; }
        type sum(int l, int r) { return sum(r) - sum(l); }
